import { hash, compare } from 'bcrypt';
import { sign, verify } from 'jsonwebtoken';
import { z } from 'zod';

// üîí Security: Input validation schema
const UserSchema = z.object({
  email: z.string().email(),
  password: z.string().min(8)
});

// ‚ö° Generated: Auth service class
export class AuthService {
  private readonly SALT_ROUNDS = 12;
  private readonly JWT_SECRET = process.env.JWT_SECRET!;

  // üéØ Optimized: Secure password hashing
  async hashPassword(password: string): Promise<string> {
    return hash(password, this.SALT_ROUNDS);
  }

  // üéØ Optimized: JWT token generation
  generateToken(userId: string): string {
    return sign(
      { sub: userId, iat: Date.now() },
      this.JWT_SECRET,
      { expiresIn: '24h' }
    );
  }

  // ‚úÖ Production-ready authentication
  async authenticate(credentials: unknown) {
    const { email, password } = UserSchema.parse(credentials);
    const user = await this.findUser(email);
    
    if (!user || !await compare(password, user.hash)) {
      throw new AuthError('Invalid credentials');
    }
    
    return { token: this.generateToken(user.id) };
  }
}
